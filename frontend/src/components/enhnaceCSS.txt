<script setup>
import { ref, onMounted, onUnmounted } from 'vue'
import { useApiStore } from '@/stores/counter'

const props = defineProps(['userId'])
const api = useApiStore()

const messages = ref([])
const newMessage = ref('')
const roomId = ref(null)
const unreadCount = ref(0)
const typingUsers = ref([])

let interval = null

async function loadRoom() {
  const res = await api.apiFetch('/api/chat/getRoom')
  roomId.value = res.room_id
}

async function loadMessages() {
  const data = await api.apiFetch(`/api/chat/getMessages/${roomId.value}`)
  
  if (messages.value.length) {
    unreadCount.value = data.length - messages.value.length
  }

  messages.value = data
}

async function sendMessage() {
  if (!newMessage.value) return

  // await api.apiPost('/chat/send', {
  //   content: newMessage.value,
  //   user_id: props.userId,
  //   room_id: roomId.value
  // })
  await api.apiFetch('/api/chat/send', 'POST', {
    content: newMessage.value,
    user_id: props.userId,
    room_id: roomId.value
  })

  newMessage.value = ''
  await loadMessages()
}

async function sendTyping(state) {
  // await api.apiPost('chat/typing', {
  //   user_id: props.userId,
  //   typing: state
  // })
  await api.apiFetch('/api/chat/typing', 'POST', {
    user_id: props.userId,
    typing: state,
  })
}

async function loadTyping() {
  typingUsers.value = await api.apiFetch('/chat/typing')
}

onMounted(async () => {
  await loadRoom()
  await loadMessages()

  interval = setInterval(() => {
    loadMessages()
    loadTyping()
  }, 2000)
})

onUnmounted(() => clearInterval(interval))
</script>

<template>
  <div class="flex flex-col h-full p-4">
    <div class="flex-1 overflow-y-auto">
      <div v-for="msg in messages" :key="msg.id">
        <strong>{{ msg.username }}:</strong>
        {{ msg.content }}
        <small>({{ msg.status }})</small>
      </div>
    </div>

    <div v-if="typingUsers.length">
      Someone is typing...
    </div>

    <div v-if="unreadCount > 0">
      {{ unreadCount }} new messages
    </div>

    <input 
      v-model="newMessage"
      @input="sendTyping(true)"
      @blur="sendTyping(false)"
      placeholder="Type message..."
    />
    <button @click="sendMessage">Send</button>
  </div>
</template>
